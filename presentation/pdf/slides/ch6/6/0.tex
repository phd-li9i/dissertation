\begin{frame}{Τι ήταν προηγουμένως αδύνατον και τώρα είναι εφικτό}

  \definecolor{g}{RGB}{0 150 0}
  \definecolor{r}{RGB}{180 0 0}

  \vspace{1cm}
  Real-time μέθοδοι \texttt{sm2} \textit{χωρίς τον υπολογισμό αντιστοιχίσεων}
  (για δισδιάστατες και πανοραμικές σαρώσεις):

  \begin{itemize}
    \item Εύρωστη βελτίωση ακρίβειας $\{$pose tracking, θέσεων προϊόντων$\}$
    \item Λύση global localisation
      \begin{itemize}
        \item Χωρίς ad hoc παραμετροποίηση (features ή μεταβλητές)
        \item Ταχύτερα από \texttt{FMI-SPOMF} ($\Rightarrow$ περισσότερες υποθέσεις $\rightarrow$ λιγότερες αποτυχίες εκτίμησης)
      \end{itemize}

  \end{itemize}

\note{\footnotesize
Αυτό λοιπόν που είναι σήμερα δυνατόν που δεν ηταν προηγουμένως είναι η ύπαρξη
  μεθόδων που λύνουν το πρόβλημα sm2 χωρίς να χρησιμοποιούν αντιστοιχίσεις, και
  που εκτελούνται σε πραγματικό χρόνο, οι οποίες είναι περισσότερο ακριβείς και
  εύρωστες από τις αντίστοιχες της βιβλιογραφίας. Οποιαδήποτε από αυτές τις
  τρεις μεθόδους είναι ικανή να βελτιώσει τη λύση του προβλήματος της
  παρατήρησης της στάσης ενός οχήματος καθώς αυτό κινείται, και να λύσει το
  πρόβλημα της εκ του μηδενός εκτίμησης της στάσης του, χωρίς να
  χρησιμοποιούνται πουθενά ad hoc μεταβλητές ή features.}


\end{frame}
